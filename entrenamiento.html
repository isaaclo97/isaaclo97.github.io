										if(used[div]==0)

bool check[20500];
queue <int> q[20500];
int resu,res;
void dfs(int x)
{
    check[x]=true;
    int size=q[x].size();
    for(int i=0;i<size;i++)
    {
        int temp=q[x].front();
        q[x].pop();
        if(!check[temp])
        {
            resu++;
            dfs(temp);
        }
    }
}

int main()
{
    int t,n,e,x,y,count;

    while(t--)
    {
        count=res=0;

        for(int i=0; i<n+1;i++) check[i]=false;
        for(int i=0;i<e;i++)
        {
            cin>>x>>y;
            q[x].push(y);

        {
            if(!check[i])
            {
                resu=1;

pair<int,int> solve(int index, int value){

    res = solve(res.first+1,value); //right node
